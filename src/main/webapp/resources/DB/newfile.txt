            // Funkcija koja pronalazi sve tostive i prikazuje ih
            function prikaziTostove() {
                var toastElList = [].slice.call(document.querySelectorAll('.toast'));
                var toastList = toastElList.map(function (toastEl) {
                    var t = new bootstrap.Toast(toastEl, { autohide: true, delay: 5000 });
                    t.show();
                    return t;
                });
            }

            // Pokreni pri učitavanju stranice
            document.addEventListener("DOMContentLoaded", prikaziTostove);

            // Ako koristiš AJAX (f:ajax), dodaj i ovo da radi nakon svakog klika:
            if (typeof jsf !== 'undefined') {
                jsf.ajax.addOnEvent(function(data) {
                    if (data.status === 'success') {
                        prikaziTostove();
                    }
                });
            }
