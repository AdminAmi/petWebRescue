<html xmlns="http://www.w3.org/1999/xhtml"
      xmlns:h="jakarta.faces.html"
      xmlns:f="jakarta.faces.core"
      xmlns:jsf="jakarta.faces"
      xmlns:c="jakarta.tags.core"
      xmlns:ui="jakarta.faces.facelets"
      xmlns:p="jakarta.faces.passthrough"
      xmlns:ezcomp="jakarta.faces.composite/ezcomp"
>
    <h:head>
        <meta name="viewport" content="width=device-width, initial-scale=1"/>
        <title>Facelet Title</title>
        <h:outputStylesheet name="bootstrap.min.css" />
        <h:outputStylesheet name="mycss.css" />
        <h:outputScript name="bootstrap.bundle.min.js" />
    </h:head>

<!-- Provjeravamo ima li ovaj input grešku u porukama -->
    <ui:param name="puniId" value="#{cc.clientId}:#{cc.attrs.ID}" />
    <ui:param name="jeNevazece" value="#{not empty facesContext.getMessageList(puniId)}" />

    <div class="form-floating mb-3 #{cc.attrs.cssDIV}">
        <input jsf:id="#{cc.attrs.ID}" 
               type="#{cc.attrs.TIP}" 
               class="form-control #{facesContext.postback ? (jeNevazece ? 'is-invalid' : 'is-valid') : ''} #{cc.attrs.CSS}" 
               placeholder="#{cc.attrs.PHOLDER}"
               jsf:value="#{cc.attrs.VALUE}" 
               jsf:required="#{cc.attrs.REQ}" 
               jsf:requiredMessage="#{cc.attrs.REQWHO}"
               disabled="#{cc.attrs.DISABLED}"/>
        
        <label for="#{puniId}">
            #{cc.attrs.PHOLDER}
        </label>
        
        <!-- Prikazuje se samo ako postoji greška -->
        <div class="invalid-feedback">
            #{cc.attrs.REQWHO}
        </div>
        <div class="valid-feedback">
            Izgleda dobro!
        </div>
    </div>

<cc:implementation>
        <ui:param name="puniId" value="#{cc.clientId}:#{cc.attrs.ID}" />
        <ui:param name="imaGresku" value="#{not empty facesContext.getMessageList(puniId)}" />

        <!-- NOVI USLOV: Zeleno je samo ako je forma poslana I ako NIJE prazno polje -->
        <!-- (Izbjegavamo zeleno na praznim opcionalnim poljima dok se stvarno ne popune) -->
        <ui:param name="prazno" value="#{empty cc.attrs.VALUE}" />

        <ui:param name="validnaKlasa" value="#{facesContext.postback and not imaGresku and not prazno ? 'is-valid' : ''}" />
        <ui:param name="invalidnaKlasa" value="#{imaGresku ? 'is-invalid' : ''}" />

        <div class="form-floating mb-3 #{cc.attrs.cssDIV}">
            <input jsf:id="#{cc.attrs.ID}" 
                   type="#{cc.attrs.TIP}"
                   class="form-control #{invalidnaKlasa} #{validnaKlasa} #{cc.attrs.CSS}" 
                   jsf:value="#{cc.attrs.VALUE}" 
                   jsf:required="#{cc.attrs.REQ}" 
                   jsf:requiredMessage="#{cc.attrs.REQWHO}"
                   disabled="#{cc.attrs.DISABLED}"/>
             <label for="#{puniId}">
            #{cc.attrs.PHOLDER}
        </label>
        
        <!-- Prikazuje se samo ako postoji greška -->
        <div class="invalid-feedback">
            #{cc.attrs.REQWHO}
        </div>
        <div class="valid-feedback">
            Izgleda dobro!
        </div>
    </div>
    </cc:implementation>
